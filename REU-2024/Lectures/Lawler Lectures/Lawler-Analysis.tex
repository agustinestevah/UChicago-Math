\chapter{Lecture 1- Measure}
Guiding examples:
\begin{enumerate}
    \item Lengths of subsets of $\bbR$
    \begin{enumerate}
        \item If $A \cap B = \emptyset,$ then $l(A) + l(B) = l(A\cup B)$ (finite additivity).
        \item $l([a,b]) = b-a.$
        \item Note that it is not true that if $\{A_\beta\}_{\beta \subset V}$ are disjoint, then \[l(\bigcup_\beta)A_\beta = \sum_{\beta}l(A_\beta)\] because $1 = l[0,1]$ but $1 \neq \sum_{x\in [0,1]}l(x)$
        \item Note that while it depends on the choice of axioms, it should generally be true that, if countable, $l(A_1 \cup A_2 \cup \dots) = \sum_{j=1}^\infty l(A_j).$
    \end{enumerate}
    \item Infinite sequences of independent coin flips
\end{enumerate}
\newcommand{\calF}{\mathcal{F}}
\defn{$\sigma-$ algebra}{A \textit{$\sigma-$ algebra,} or a \textit{$\sigma$ field} of subsets of a set $X$ is a collection of subsets satisfying:
\begin{enumerate}
    \item $\emptyset\in \calF.$
    \item If $A \in \calF,$ then $X\sm A \in \calF.$
    \item If $A_1, A_2, \dots, \in \calF,$ then $\bigcup_{n=1}^\infty A_n \in \calF.$
\end{enumerate}}

\defn{Measure}{A (non-negative) \textit{measure} is a function $\mu:\calF \to [0,\infty]$ such that $\mu(\emptyset) = 0$ and it has countable additivity (c above).}
\rmkb{We define a \textit{measure space} as $(X,\calF)$ and a \textit{measurable space} as $(X,\calF, \mu).$}
\fact{In Probability, the equivalent is of the set $X$ is $\Omega,$ or the set of outcomes, $P,$ the probability is $\mu,$ and $P(\Omega) = 1.$}

\lem{Collection of $\sigma-$algebras}{If $\{\calF_\beta\}$ is a collection of $\sigma-$ algebra of $X,$ then $\bigcap_{\calF} X_\beta$ is a $\sigma-$ algebra}
\pf{:\\\begin{enumerate}
    \item If $A\in \bigcap_{\beta} \calF_\beta,$ then $A \in \calF_\beta$ for all $\beta,$ and so by definition, $X\sm A \in \calF_\beta$ for all $\beta,$ and thus $X\sm A \in \bigcap_\beta\calF_\beta.$
    \item If $A_1, A_2, \dots \in \bigcap_{\beta}\calF_\beta,$ then $A_1, A_2, \dots \in \calF_\beta$ for all $\beta,$ then $\bigcup_{n=1}^\infty A_n \in \calF_\beta$ for all $\beta$, and so $\bigcup_{n=1}^na_\beta \in \bigcap_{\beta}\calF$
\end{enumerate}
}

\defn{Borel Sets}{A collection of subsets, $\alpha,$ then \[\calF_\alpha = \text{smallest $\sigma-$algebra containing all the sets in $\alpha$} = \bigcap_{\calF \sigma-\text{algebra},} \calF_\beta\]}
